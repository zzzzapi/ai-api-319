Node.js接入示例


const axios = require('axios');

const API_KEY = "您的API密钥";
const API_URL = "https://api.zzzzapi.com/v1/chat/completions";

async function callOpenAI() {
  try {
    const response = await axios.post(
      API_URL,
      {
        model: "gpt-4o",
        messages: [
          {role: "system", content: "你是智能助手"},
          {role: "user", content: "介绍OpenAI API的功能"}
        ]
      },
      {
        headers: {
          'Authorization': `Bearer ${API_KEY}`,
          'Content-Type': 'application/json'
        }
      }
    );
    console.log(response.data);
  } catch (error) {
    console.error(error);
  }
}

callOpenAI();
